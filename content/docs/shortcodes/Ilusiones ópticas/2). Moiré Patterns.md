  p5.setup = function() {
    p5.createCanvas(600, 600);
  };

  p5.draw = function() {
    p5.background(200);
    p5.stroke(36, 90, 49);
    p5.strokeWeight(6);
    p5.noFill();

    for(diameter=20; diameter<=580; diameter +=25){
      p5.circle(300, 300, diameter);
    }

    p5.stroke(255);
    p5.strokeWeight(6);
    p5.stroke(0, 0, 255);

    for(diameter2=7; diameter2<=580; diameter2 +=25){
      p5.circle(p5.mouseX, p5.mouseY, diameter2);
    }
  }
